\section{The Ramanujan Tau Bridge}
\label{sec:tau}

Let $\tau\colon\mathbb{Z}_{>0}\to\mathbb{Z}$ be the Ramanujan tau function,
defined by the generating series
\[
  \Delta(q) = q\prod_{n=1}^{\infty}(1-q^n)^{24} = \sum_{n=1}^{\infty}\tau(n)q^n,
  \qquad \tau(1)=1,\;
  \tau(2)=-24,\;
  \tau(3)=252,\;
  \tau(5)=4830,\;
  \tau(7)=-16744.
\]
We show that the first two nontrivial values $\tau(2)$ and $\tau(3)$ are encoded
directly in the spectral parameters of $W(3,3)$.

\begin{theorem}[Tau bridge]\label{thm:tau}
With the notation of Section~\ref{sec:uniqueness},
\[
  \tau(2) = -f(3) = -24, \qquad \tau(3) = k(3)\cdot 3\cdot\Phi_6(3) = 252.
\]
\end{theorem}
\begin{proof}
$-f(3) = -q(q+1)^2/2\big|_{q=3} = -24$.
And $k(3)\cdot 3\cdot\Phi_6(3)=12\cdot 3\cdot 7=252$.
\end{proof}

\begin{remark}[Hecke recursion]
The Hecke recursion $\tau(p^{a+1})=\tau(p)\tau(p^a)-p^{11}\tau(p^{a-1})$
at $p=k-1=11$ involves the shift $p^{11}=(k-1)^{11}$, connecting the
weight-$12$ structure of $\Delta$ to the Ramanujan degree $k-1=11$.
\end{remark}

\begin{theorem}[$\Phi_6$ barrier]\label{thm:barrier}
For primes $p\equiv 1\pmod{\Phi_6(3)=7}$, Ramanujan's congruence
$\tau(p)\equiv 0\pmod{p}$ holds.
The first such prime is $p=7=\Phi_6(3)$ itself, followed by $p=29,43,71,\ldots$
\end{theorem}

\begin{theorem}[C5 via Frobenius]\label{thm:C5}
Let $E$ be the elliptic curve
\[
  E\colon y^2 = x^3 - 35x + 98,
\]
which has $j(E)=-3375=-g^3$ and conductor $N=49$.
Then
\[
  a_{k-1}(E) = a_{11}(E) = -(q+1) = \mathrm{ev}_s.
\]
Moreover, $E$ is a CM curve with CM by $\mathbb{Z}[(1+\sqrt{-7})/2]$,
and the Frobenius traces satisfy:
\[
\begin{array}{lcl}
  a_p(E) = 0 & & \text{if } \left(\tfrac{-7}{p}\right)=-1\ (\text{$p$ inert})\\
  a_{11}(E) = -4 = \mathrm{ev}_s && \text{(C5)}\\
  a_{23}(E) = -8,\quad a_{29}(E) = 2 = \mathrm{ev}_r && \text{(split primes)}
\end{array}
\]
The discriminant check $a_{11}^2-4\cdot 11 = 16-44=-28=-4\Phi_6$
identifies the $j(-7)=-g^3$ tower entry as the CM locus of $p_2(u)$.
\end{theorem}
\begin{proof}
Direct point-counting over $\mathbb{F}_{11}$ gives $\#E(\mathbb{F}_{11})=16$,
so $a_{11}=11+1-16=-4=\mathrm{ev}_s$. The CM property ensures $a_p=0$
for all inert $p$, verified for $p\in\{3,5,13,17,19\}$.
The curve is the $(-1)$-quadratic twist of $y^2=x^3-35x-98$, both having $j=-3375$.
\end{proof}

\subsection*{The j-tower and its cube structure}

The Ihara zeta function of $W(3,3)$ factors through two quadratic polynomials
whose CM discriminants are $-4\Phi_6$ and $-4\Phi_4$:
\[
  \mathrm{disc}(p_2) = -4\Phi_6 = -28,\qquad
  \mathrm{disc}(p_1) = -4\Phi_4 = -40.
\]
The field $\mathbb{Q}(\sqrt{-\Phi_6})=\mathbb{Q}(\sqrt{-7})$ has class number $h=1$
(Heegner), while $\mathbb{Q}(\sqrt{-\Phi_4})=\mathbb{Q}(\sqrt{-10})$ has $h=2$.
The $s$-eigenspace sector therefore generates a complete tower
of $j$-invariants that are perfect cubes of $W(3,3)$ spectral invariants.
Let $\Sigma = 2^{g/3}=2^5=32$ (a spectral 2-power, valid since $3\mid g=15$)
and $P = q(\Phi_4/2)(\Phi_4+\Phi_6)=255$. Then:

\begin{equation}\label{eq:jtower}
\renewcommand{\arraystretch}{1.3}
\begin{array}{rclll}
  j(D=-4)  &=& k^3 = 1728     & &\mathbb{Z}[i]\\
  j(D=-7)  &=& -g^3 = -3375   & &\text{maximal order of }\mathbb{Q}(\sqrt{-7})\\
  j(D=-8)  &=& (v/2)^3 = 8000 & &\mathbb{Z}[\sqrt{-2}]\\
  j(D=-11) &=& -\Sigma^3 = -32768  & &\text{maximal order of }\mathbb{Q}(\sqrt{-11})\\
  j(D=-28) &=& P^3=16{,}581{,}375 & \quad P=q\cdot\tfrac{\Phi_4}{2}\cdot(\Phi_4+\Phi_6),
              & \mathbb{Z}[\sqrt{-7}],\;f=2.
\end{array}
\end{equation}

\begin{proposition}[Cube structure]\label{prop:cube}
The five cube roots $\{k,\, g,\, v/2,\, \Sigma,\, P\}$ are
spectral invariants of $W(3,3)$. No Heegner discriminant with
$|D|>4\Phi_6=28$ has its cube root in this set.
Equivalently, the $j$-tower is the complete intersection of:
\begin{enumerate}
\item[(a)] Heegner discriminants satisfying $|D|\leq 4\Phi_6$;
\item[(b)] the cube root lies in the $W(3,3)$ spectral invariant set.
\end{enumerate}
\end{proposition}

\noindent
In contrast, the $r$-eigenspace sector ($h=2$, disc $-40$) has $j$-values
approximately $4.3\times 10^8$ and $2.1\times 10^4$ that do not appear
as cubes of $W(3,3)$ parameters, confirming that the $j$-tower is entirely
an $s$-sector phenomenon.
