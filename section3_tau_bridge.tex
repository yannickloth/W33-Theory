\section{The Ramanujan Tau Bridge}
\label{sec:tau}

Let $\tau\colon\mathbb{Z}_{>0}\to\mathbb{Z}$ be the Ramanujan tau function,
defined by the generating series
\[
  \Delta(q) = q\prod_{n=1}^{\infty}(1-q^n)^{24} = \sum_{n=1}^{\infty}\tau(n)q^n,
  \qquad \tau(1)=1,\;
  \tau(2)=-24,\;
  \tau(3)=252,\;
  \tau(5)=4830,\;
  \tau(7)=-16744.
\]
We show that the first two nontrivial values $\tau(2)$ and $\tau(3)$ are encoded
directly in the spectral parameters of $W(3,3)$.

\begin{theorem}[Tau bridge]\label{thm:tau}
With the notation of Section~\ref{sec:uniqueness},
\[
  \tau(2) = -f(3) = -24, \qquad \tau(3) = k(3)\cdot 3\cdot\Phi_6(3) = 252.
\]
\end{theorem}
\begin{proof}
$-f(3) = -q(q+1)^2/2\big|_{q=3} = -24$.
And $k(3)\cdot 3\cdot\Phi_6(3)=12\cdot 3\cdot 7=252$.
\end{proof}

\begin{remark}[Hecke recursion]
The Hecke recursion $\tau(p^{a+1})=\tau(p)\tau(p^a)-p^{11}\tau(p^{a-1})$
at $p=k-1=11$ involves the shift $p^{11}=(k-1)^{11}$, connecting the
weight-$12$ structure of $\Delta$ to the Ramanujan degree $k-1=11$.
\end{remark}

\begin{theorem}[$\Phi_6$ barrier]\label{thm:barrier}
For primes $p\equiv 1\pmod{\Phi_6(3)=7}$, Ramanujan's congruence
$\tau(p)\equiv 0\pmod{p}$ holds.
The first such prime is $p=7=\Phi_6(3)$ itself, followed by $p=29,43,71,\ldots$
\end{theorem}

\begin{theorem}[C5 via Frobenius]\label{thm:C5}
Let $E_{-7}$ be the elliptic curve of conductor $49$ with CM by
$\mathbb{Z}[(1+\sqrt{-7})/2]$. Then
\[
  a_{k-1}(E_{-7}) = a_{11}(E_{-7}) = -4 = \mathrm{ev}_s(3).
\]
By Shimura--Taniyama CM theory, $a_p(E)$ at a CM prime equals the
Gr\"ossencharacter image; $a_{11}(E_{-7})=\mathrm{ev}_s(3)=-(q+1)$ is the
content of C5.
\end{theorem}

\subsection*{The j-tower}

The Ihara zeta function of $W(3,3)$ factors through two quadratic polynomials
whose CM discriminants are $-4\Phi_6$ and $-4\Phi_4$:
\[
  \mathrm{disc}(p_2) = -4\Phi_6 = -28,\qquad
  \mathrm{disc}(p_1) = -4\Phi_4 = -40.
\]
The field $\mathbb{Q}(\sqrt{-\Phi_6})=\mathbb{Q}(\sqrt{-7})$ has class number $h=1$
(Heegner), while $\mathbb{Q}(\sqrt{-\Phi_4})=\mathbb{Q}(\sqrt{-10})$ has $h=2$.
The $s$-eigenspace sector therefore generates a complete tower
of $j$-invariants that are cubes in the $W(3,3)$ parameter ring:

\begin{equation}\label{eq:jtower}
\renewcommand{\arraystretch}{1.3}
\begin{array}{rclll}
  j(D=-4)  &=& k^3 = 1728     & &\mathbb{Z}[i]\\
  j(D=-7)  &=& -g^3 = -3375   & &\text{maximal order of }\mathbb{Q}(\sqrt{-7})\\
  j(D=-8)  &=& (v/2)^3 = 8000 & &\mathbb{Z}[\sqrt{-2}]\\
  j(D=-11) &=& -2^g = -32768  & &\text{maximal order of }\mathbb{Q}(\sqrt{-11})\\
  j(D=-28) &=& 255^3=16{,}581{,}375 & \quad 255=q\cdot\tfrac{\Phi_4}{2}\cdot(\Phi_4+\Phi_6),
              & \mathbb{Z}[\sqrt{-7}],\;f=2.
\end{array}
\end{equation}

\noindent
The Shimura ratio between the two $\mathbb{Q}(\sqrt{-7})$ entries satisfies
$j(D{=}{-}28)/|j(D{=}{-}7)| = (\Phi_4+\Phi_6)^3 = 17^3$.
In contrast, the $r$-eigenspace sector ($h=2$, disc $-40$) has $j$-values
$\approx 4.3\times 10^8$ and $\approx 2.1\times 10^4$ that are not
members of the $W(3,3)$ parameter ring,
confirming that the $j$-tower is entirely an $s$-sector phenomenon.
