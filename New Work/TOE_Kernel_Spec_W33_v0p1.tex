\section*{Purpose}
This document compresses the \emph{verified} structural results from our W33/Witting work into a single coherent architecture: keys, authenticated transport, curvature-as-syndrome, phase clocking, and an E6 firewall carving a code subspace.

\section{Primitive objects}

\subsection{Universe $\mathcal{U}$}
A finite information geometry with \textbf{40 atomic states} $S=\{0,\dots,39\}$.

\subsection{Orthogonality graph $G_{\perp}$}
A strongly regular graph on $S$ with parameters $\mathrm{SRG}(40,12,2,4)$.
\begin{itemize}
  \item \textbf{Orthogonal / perfectly hidden:} $u\perp v$ iff $(u,v)$ is an edge in $G_{\perp}$.
  \item \textbf{Nonorth / potentially coupled:} $u\not\perp v$ otherwise.
\end{itemize}
\noindent\textbf{Encryption axiom:} perfect secrecy = orthogonality (zero overlap); coupling requires nonorthogonality (partial key match).

\section{Encapsulation: 10 objects with 2-bit private keyspaces}
Partition $S$ into \textbf{10 blocks} (“objects/contexts”) $B_0,\dots,B_9$, each of size 4.
Each state $s\in B_i$ carries a \textbf{2-bit internal key}
\[
x(s)\in \mathbb{Z}_2^2=\{00,01,10,11\}.
\]
\noindent\textbf{Pauli-frame / key interpretation:}
$x=(a,b)\leftrightarrow X^a Z^b$ (Pauli exponent key), and $11$ corresponds to $XZ\sim iY$ (composite/fermion channel).

\section{Authenticated interaction channel: N180 as unique handshake}
Define an authenticated interaction relation $N_{180}\subset S\times S$ with the \textbf{unique handshake property}:

For any ordered block pair $(A,B)$ and any $x\in\mathbb{Z}_2^2$, there exists a \textbf{unique} partner key
\[
T_{AB}(x)\in\mathbb{Z}_2^2
\]
such that the corresponding states interact:
\[
s=(A,x),\quad s'=(B,T_{AB}(x))\quad\Rightarrow\quad (s,s')\in N_{180}.
\]
Thus each ordered pair admits a \textbf{unique key-exchange map} $T_{AB}$.

\section{Transport field: affine gauge connection on keys}
Each $T_{AB}$ is affine:
\[
T_{AB}(x)=M_{AB}x+t_{AB},\qquad (M_{AB},t_{AB})\in AGL(2,2).
\]
\begin{itemize}
  \item $t_{AB}\in\mathbb{Z}_2^2$ is a Pauli translation (session-key translation).
  \item $M_{AB}\in GL(2,2)$ is a Clifford-router action on Pauli exponents.
\end{itemize}
\noindent\textbf{Field A (transport/gauge):} the connection $(M_{AB},t_{AB})$.

\section{Curvature / syndrome: holonomy on triples is a parity-check design}
For any block triple $(A,B,C)$, define holonomy
\[
H_{ABC}=T_{CA}\circ T_{BC}\circ T_{AB}\in AGL(2,2).
\]
Its \textbf{matrix class} (identity vs swap) is an orientation-independent $\mathbb{Z}_2$ syndrome on unordered triples.

Unordered triples split $60/60$, and the “odd” triples form a \textbf{$2$-$(10,3,4)$ design}:
\begin{itemize}
  \item 60 triples of size 3,
  \item every pair of blocks occurs in exactly 4 odd triples.
\end{itemize}
\noindent\textbf{Field B (syndrome/curvature):} the $\mathbb{Z}_2$ holonomy class on triangles (a MAC/parity-check structure).

\section{Phase field: quantized interference clock}
Each directed interaction edge carries a \textbf{phase tick}
\[
k_6(A\to B;x)\in \mathbb{Z}_6
\]
derived from gauge-invariant Bargmann/overlap phases.

This yields a local session key / clock per interaction:
\[
(\Delta x,k_6)\in \mathbb{Z}_2^2\times \mathbb{Z}_6\qquad\text{(24-state clock)}.
\]
\noindent\textbf{Field C (phase/U(1)-like):} the $\mathbb{Z}_6$ phase 1-form and its loop holonomies.

\section{E6 kernel: firewall / admissibility constraint}
Pick an embedding vertex $v_0\in S$. Let $H_{12}=N(v_0)$ (12 orth neighbors) and let $H_{27}$ be the 27 non-neighbors of $v_0$.

For any nonorth pair $\{u,v\}\subset H_{27}$ define the witness set
\[
W(u,v)=N(u)\cap N(v)\qquad (\lvert W(u,v)\rvert=4).
\]

\subsection*{Firewall rule (theorem-level)}
\[
\boxed{\mathrm{bad}(u,v)\iff W(u,v)\subset H_{12}.}
\]
The allowed E6 kernel edges are the complement:
\[
S_{\text{Schl\"afli}}=\mathrm{nonorth}(H_{27})\setminus \mathrm{bad}.
\]

\subsection*{Composite-channel prohibition (corollary)}
On \textbf{bad} edges, the Pauli translation channel $t=11$ (i.e. $XZ\sim iY$) never occurs.

\noindent\textbf{Interpretation:}
$W(u,v)\subset H_{12}$ is a \emph{local-only transcript} (vulnerable handshake). The firewall deletes those transitions and forbids the composite/fermion channel in vulnerable configurations.

\section{Sectors (“particle types”): idempotent decomposition of coupling algebra}
A rank-6 association scheme yields idempotents $E_0,\dots,E_5$ defining sector planes. From selection-rule deltas we extract three effective modes:
\begin{itemize}
  \item $G$: authenticated router/gauge-like mode,
  \item $M$: endpoint/matter-like mode,
  \item $C$: constraint/firewall mode.
\end{itemize}
Sectors carry charges under $(G,M,C)$; couplings are low-rank products of charges (effective field theory on sectors).
